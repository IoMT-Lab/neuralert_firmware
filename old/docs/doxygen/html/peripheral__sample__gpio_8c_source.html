78static int set_gpio_interrupt(HANDLE handler, UINT8 pin_num, UINT8 int_type, UINT8 int_pol, void *callback_func)
79{
80 UINT16 pin, int_en_status;
81 UINT32 ioctldata[3];
82 int ret;
83
84 if (15 < pin_num ) {
85 return FALSE;
86 }
87
88 if (handler == NULL) {
89 return FALSE;
90 }
91
92 pin = 0x01 << pin_num;
93 ret = GPIO_IOCTL(handler, GPIO_SET_INPUT, &pin);
94
95 ret = GPIO_IOCTL(handler, GPIO_GET_INTR_MODE, &ioctldata[0]);
96 /* interrupt type 1: edge, 0: level*/
97 ioctldata[0] &= ~(1 << pin_num);
98 ioctldata[0] |= (int_type << pin_num);
99 /* interrupt pol 1: high active, 0: low active */
100 ioctldata[1] &= ~(1 << pin_num);
101 ioctldata[1] |= (int_pol << pin_num);
102 ret = GPIO_IOCTL(handler, GPIO_SET_INTR_MODE, &ioctldata[0]);
103
104 /* register callback function */
105 ioctldata[0] = pin; /* interrupt pin */
106 ioctldata[1] = (UINT32) callback_func; /* callback function */
107 ioctldata[2] = (UINT32) pin_num; /* param data */
108 ret = GPIO_IOCTL(handler, GPIO_SET_CALLACK, ioctldata);
109
110 ret = GPIO_IOCTL(handler, GPIO_GET_INTR_ENABLE, &int_en_status);
111 int_en_status |= pin;
112 ret = GPIO_IOCTL(handler, GPIO_SET_INTR_ENABLE, &int_en_status);
113
114 return ret;
115}

158void peripheral_sample_gpio(void *param)
159{
160 DA16X_UNUSED_ARG(param);
161
162 HANDLE gpio;
163 UINT32 pin, toggle = 0, loop_cnt = 0;;
164 UINT16 read_data, write_data;
165
166 /*
167 * board init
168 */
169 Board_initialization();
170
171 /*
172 * GPIOA port init
173 */
174 gpio = GPIO_CREATE(GPIO_UNIT_A);
175 GPIO_INIT(gpio);
176
177 /*
178 * GPIOA[0],GPIOA[4] output high low toggle
179 */
180 pin = GPIO_PIN0 | GPIO_PIN4;
181 GPIO_IOCTL(gpio, GPIO_SET_OUTPUT, &pin);
182
183 /*
184 * GPIOA[1] input
185 */
186 pin = GPIO_PIN1;
187 GPIO_IOCTL(gpio, GPIO_SET_INPUT, &pin);
188
189#if PAD_PULL_CONTROL
190 /*
191 * GPIOA[1] input pull control it can make gpio pad pull up or pull down or HIZ
192 */
193 _da16x_gpio_set_pull(GPIO_UNIT_A, GPIO_PIN1, PULL_UP);
194 /* or */
195 _da16x_gpio_set_pull(GPIO_UNIT_A, GPIO_PIN1, PULL_DOWN);
196 /* or */
197 _da16x_gpio_set_pull(GPIO_UNIT_A, GPIO_PIN1, HIGH_Z);
198#endif
199
200 /*
201 * GPIOA[2] interrupt active low , Edge trigger
202 */
203 set_gpio_interrupt(gpio, 2, GPIO_INT_TYPE_EDGE, GPIO_INT_POL_LOW, (void *)gpio_callback );
204
205 /*
206 * GPIOA[3] interrupt active high , Edge trigger
207 */
208 set_gpio_interrupt(gpio, 3, GPIO_INT_TYPE_EDGE, GPIO_INT_POL_HIGH, (void *)gpio_callback );
209
210#ifdef GPIOC6_BOTH_EDGE_INTERRUPT
211 /*
212 * GPIOC port init
213 */
214 gpioc = GPIO_CREATE(GPIO_UNIT_C);
215 GPIO_INIT(gpioc);
216 /*
217 * GPIOC[6] interrupt Both edge
218 */
219 pin = GPIO_PIN6;
220 GPIO_IOCTL(gpioc, GPIO_SET_INPUT, &pin);
221
222 GPIO_READ(gpioc, GPIO_PIN6, &read_data, sizeof(UINT16));
223
224 set_gpio_interrupt(gpioc, 6, GPIO_INT_TYPE_EDGE, !(GPIO_PIN6 & read_data), (void *)gpioc_callback );
225#endif
226#ifdef RTC_GPO_CONTROL
227 RTC_GPO_OUT_INIT(1); // 0: auto, 1: manual
228#endif
229 while (1) {
230 if (toggle) {
231 /* GPIOA[0],GPIOA[4] to high */
232 write_data = GPIO_PIN0 | GPIO_PIN4;
233 GPIO_WRITE(gpio, GPIO_PIN0 | GPIO_PIN4, &write_data, sizeof(UINT16));
234#ifdef RTC_GPO_CONTROL
235 RTC_GPO_OUT_CONTROL(toggle);
236#endif
237 toggle = 0;
238 } else {
239 /* GPIOA[0],GPIOA[4] to low*/
240 write_data = 0;
241 GPIO_WRITE(gpio, GPIO_PIN0 | GPIO_PIN4, &write_data, sizeof(UINT16));
242#ifdef RTC_GPO_CONTROL
243 RTC_GPO_OUT_CONTROL(toggle);
244#endif
245 toggle = 1;
246 }
247
248 GPIO_READ(gpio, GPIO_PIN1, &read_data, sizeof(UINT16));
249
250 PRINTF("GPIOA[0] output %s, GPIOA[4] output %s, GPIOA[1] input %s\n",
251 (write_data & GPIO_PIN0) ? "high" : "low",
252 (write_data & GPIO_PIN4) ? "high" : "low",
253 (read_data & GPIO_PIN1) ? "high" : "low");
254
255 vTaskDelay(100); // 1000ms sleep
256
257 if (++loop_cnt == 30) {
258 PRINTF("Clear GPIO interrupt \n");
259 /* Disable GPIO2,3 interrupt */
260 clear_gpio_interrupt(gpio, 2);
261 clear_gpio_interrupt(gpio, 3);
262 }
263 }
264 GPIO_CLOSE(gpio);
265}
