78static UINT32 watchdog_active(WDOG_FSM_TYPE *wfsm, UINT32 wdogcount)
79{
80 if ( wfsm != NULL ) {
81 if ( xSemaphoreTake((wfsm->mutex), portMAX_DELAY) == pdTRUE ) {
82
83 WDOG_IOCTL(wfsm->watchdog, WDOG_SET_DISABLE, NULL );
84
85 WDOG_IOCTL(wfsm->watchdog, WDOG_SET_COUNTER, &wdogcount ); // update
86
87 wfsm->currtx = (TaskHandle_t)xTaskGetCurrentTaskHandle();
88
89 WDOG_IOCTL(wfsm->watchdog, WDOG_SET_ENABLE, NULL );
90
91 return TRUE;
92 }
93 }
94
95 return FALSE;
96}

98static void watchdog_deactive(WDOG_FSM_TYPE *wfsm)

104
105 WDOG_IOCTL(wfsm->watchdog, WDOG_SET_DISABLE, NULL );

111void WatchDog_sample(void *param)

114 extern void wdt_enable_auto_goto_por();

117 /* DO SOMETHING */

119 UINT32 idx;

123 PRINTF("WATCHDOG_SAMPLE\n");
124 PRINTF("The system may reboot repeatedly while running this code.\n");
125 PRINTF("If you don't want it, type \"reset\" to stop it.\n");

127 vTaskDelay(500); // 5 sec

129 PRINTF(">>> WatchDog Test...\n" );
130 PRINTF(" >> set watchdog timeout : %d sec \n", WDT_TIMEOUT);
131 set_wdt_rescale_time(WDT_TIMEOUT); // set rescale time (2 sec)
132 wdog_fsm = (WDOG_FSM_TYPE *)pvPortMalloc(sizeof(WDOG_FSM_TYPE));
133 if ( wdog_fsm == NULL ) {
134 PRINTF(" [%s] alloc error\n", __func__);

136 }

138 // WDOG Creation
139 wdog_fsm->watchdog = WDOG_CREATE(WDOG_UNIT_0) ;

147 // WDOG Initialization
148 WDOG_INIT(wdog_fsm->watchdog);

154 ioctldata[0] = 0; // index
155 ioctldata[1] = (UINT32)&sample_watchdog_callback; // callback

164 _sys_clock_read(&cursysclock, sizeof(UINT32));

166 vTaskDelay(100); // 1 sec

171
172 wdogcont = (cursysclock * WDT_TIMEOUT); // 2 sec
173
174 watchdog_active(wdog_fsm, wdogcont);
175
176 PRINTF("*** %d'th Test ************\n", idx);
177 vTaskDelay(10); // 100 msec
178
179 PRINTF(" >>> Start LOOP (loop count: %d MEGA): \n", idx * 2);
180 vTaskDelay(10); // 100 msec
181 loop_count = 0;
182 while (1) {
183 if (!(++loop_count % (idx * 2 * MEGA))) {
184 vTaskDelay(10); // 100 msec
185 break;
186 }
187 asm volatile ("nop \n");
188 }

190
191 watchdog_deactive(wdog_fsm);
192 vTaskDelay(100);
193 }
194
195 vSemaphoreDelete(wdog_fsm->mutex);
196 wdog_fsm->mutex = NULL;
197
198 // WDOG Uninitialization
199 WDOG_CLOSE(wdog_fsm->watchdog);
200
201 vPortFree(wdog_fsm);
202
203 vTaskDelete(NULL);
204
205 return ;
206}
