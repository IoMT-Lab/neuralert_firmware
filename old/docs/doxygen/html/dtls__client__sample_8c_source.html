114int dtls_client_sample_timer_get_state(void *ctx)
115{
116 dtls_client_sample_timer_t *timer = NULL;
117 unsigned int elapsed = 0;
118
119 if (!ctx) {
120 PRINTF("[%s] Invaild paramter\n", __func__);
121 return CANCELLED;
122 }
123
124 timer = (dtls_client_sample_timer_t *)ctx;
125
126 if (timer->fin_ms == 0) {
127 return CANCELLED;
128 }
129
130 elapsed = xTaskGetTickCount() - timer->snapshot;
131
132 if (elapsed >= pdMS_TO_TICKS(timer->fin_ms)) {
133 return FIN_EXPIRY;
134 }
135
136 if (elapsed >= pdMS_TO_TICKS(timer->int_ms)) {
137 return INT_EXPIRY;
138 }
139
140 return NO_EXPIRY;
141}

143void dtls_client_sample(void *param)
144{

147 const char *pers = "dtls_client";
148 int ret = 0;
149 char data_buffer[DTLS_CLIENT_SAMPLE_DEF_BUF_SIZE] = {0x00,};
150 size_t len = sizeof(data_buffer);
151 char str_port[8] = {0x00,};
152
153 mbedtls_net_context server_ctx;
154 mbedtls_ssl_context ssl_ctx;
155 mbedtls_ssl_config ssl_conf;
156 mbedtls_ctr_drbg_context ctr_drbg;
157 mbedtls_entropy_context entropy;
158 dtls_client_sample_timer_t timer;
159
160 PRINTF("[%s] Start of DTLS Client sample\n", __func__);
161
162 //Init session
163 mbedtls_net_init(&server_ctx);
164
165 //Init SSL context
166 mbedtls_ssl_init(&ssl_ctx);
167
168 //Init SSL config
169 mbedtls_ssl_config_init(&ssl_conf);
170
171 //Init CTR-DRBG context
172 mbedtls_ctr_drbg_init(&ctr_drbg);
173
174 //Init Entropy context
175 mbedtls_entropy_init(&entropy);
176
177 memset(&timer, 0x00, sizeof(dtls_client_sample_timer_t));
178
179 PRINTF("\nConnecting to udp/%s:%d...\n",
180 DTLS_CLIENT_SAMPLE_DEF_SERVER_IP_ADDR,
181 DTLS_CLIENT_SAMPLE_DEF_SERVER_PORT);
182
183 snprintf(str_port, sizeof(str_port), "%d", DTLS_CLIENT_SAMPLE_DEF_SERVER_PORT);
184
185 ret = mbedtls_net_connect(&server_ctx,
186 DTLS_CLIENT_SAMPLE_DEF_SERVER_IP_ADDR, str_port,
187 MBEDTLS_NET_PROTO_UDP);
188 if (ret) {
189 PRINTF("\n[%s] Failed to connect DTLS session(0x%x)\n", __func__, -ret);
190 goto end_of_task;
191 }
192
193 PRINTF("ok\n");
194
195 PRINTF("Setting up the DTLS structure...\n");
196
197 ret = mbedtls_ssl_config_defaults(&ssl_conf,
198 MBEDTLS_SSL_IS_CLIENT,
199 MBEDTLS_SSL_TRANSPORT_DATAGRAM,
200 MBEDTLS_SSL_PRESET_DEFAULT);
201 if (ret) {
202 PRINTF("\n[%s] Failed to init dtls config(0x%x)\n", __func__, -ret);
203 goto end_of_task;
204 }
205
206 ret = mbedtls_ctr_drbg_seed(&ctr_drbg, mbedtls_entropy_func, &entropy,
207 (const unsigned char *)pers, strlen(pers));
208 if (ret) {
209 PRINTF("\n[%s] Failed to set CTR-DRBG(0x%x)\n", __func__, -ret);
210 goto end_of_task;
211 }
212
213 mbedtls_ssl_conf_rng(&ssl_conf, mbedtls_ctr_drbg_random, &ctr_drbg);
214 mbedtls_ssl_conf_authmode(&ssl_conf, MBEDTLS_SSL_VERIFY_NONE);
215 mbedtls_ssl_conf_dtls_anti_replay(&ssl_conf, MBEDTLS_SSL_ANTI_REPLAY_ENABLED);
216 mbedtls_ssl_conf_read_timeout(&ssl_conf, DTLS_CLIENT_SAMPLE_DEF_TIMEOUT);
217 mbedtls_ssl_conf_handshake_timeout(&ssl_conf,
218 DTLS_CLIENT_SAMPLE_DEF_HANDSHAKE_MIN_TIMEOUT,
219 DTLS_CLIENT_SAMPLE_DEF_HANDSHAKE_MAX_TIMEOUT);
220
221 ret = mbedtls_ssl_setup(&ssl_ctx, &ssl_conf);
222 if (ret) {
223 PRINTF("\n[%s] Failed to setup dtls contex(0x%x)\n", __func__, -ret);
224 goto end_of_task;
225 }
226
227 mbedtls_ssl_set_bio(&ssl_ctx, &server_ctx,
228 mbedtls_net_send, NULL, mbedtls_net_recv_timeout);
229
230 mbedtls_ssl_set_timer_cb(&ssl_ctx, &timer,
231 dtls_client_sample_timer_start,
232 dtls_client_sample_timer_get_state);
233
234 PRINTF("ok\n");
235
236 PRINTF("Performing the DTLS handshake...\n");
237
238 while ((ret = mbedtls_ssl_handshake(&ssl_ctx)) != 0) {
239 if (ret == MBEDTLS_ERR_NET_CONN_RESET) {
240 PRINTF("\n[%s] Peer closed the connection(0x%x)\n", __func__, -ret);
241 goto end_of_task;
242 }
243
244 if ((ret != MBEDTLS_ERR_SSL_WANT_READ) && (ret != MBEDTLS_ERR_SSL_WANT_WRITE)) {
245 PRINTF("\n[%s] Failed to do dtls handshake(0x%x)\n", __func__, -ret);
246 goto end_of_task;
247 }
248 }
249
250 PRINTF("ok\n");
251
252 //Data transmit
253 do {
254
255 len = sizeof(data_buffer) - 1;
256 memset(data_buffer, 0x00, sizeof(data_buffer));
257
258 PRINTF("< Read from server: ");
259
260 ret = mbedtls_ssl_read(&ssl_ctx, (unsigned char *)data_buffer, len);
261 if (ret <= 0) {
262 switch (ret) {
263 case MBEDTLS_ERR_SSL_WANT_READ:
264 case MBEDTLS_ERR_SSL_WANT_WRITE:
265 PRINTF("\nNeed more data - mbedtls_ssl_read(0x%x)\n", -ret);
266 continue;
267 case MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY:
268 PRINTF("\nConnection was closed gracefully\n");
269 goto end_of_task;
270 case MBEDTLS_ERR_NET_CONN_RESET:
271 PRINTF("\nConnection was reset by peer\n");
272 goto end_of_task;
273 default:
274 PRINTF("\nFailed to read data(0x%x)\n", -ret);
275 break;
276 }
277 goto end_of_task;
278 }
279
280 len = ret;
281
282 PRINTF("%d bytes read\n", len);
283
284 dtls_client_sample_hex_dump("Received data", (unsigned char *)data_buffer, len);
285
286 PRINTF("> Write to server: ");
287
288 while ((ret = mbedtls_ssl_write(&ssl_ctx, (unsigned char *)data_buffer, len)) <= 0) {
289 switch (ret) {
290 case MBEDTLS_ERR_SSL_WANT_READ:
291 case MBEDTLS_ERR_SSL_WANT_WRITE:
292 PRINTF("\nNeed more data - mbedtls_ssl_write(0x%x)\n", -ret);
293 continue;
294 case MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY:
295 PRINTF("\nConnection was closed gracefully\n");
296 goto end_of_task;
297 case MBEDTLS_ERR_NET_CONN_RESET:
298 PRINTF("\nConnection was reset by peer\n");
299 goto end_of_task;
300 default:
301 PRINTF("\nFailed to write data(0x%x)\n", -ret);
302 break;
303 }
304 goto end_of_task;
305 }
306
307 PRINTF("%d bytes written\n", len);
308
309 dtls_client_sample_hex_dump("Sent data", (unsigned char *)data_buffer, len);
310 } while (1);
311
312end_of_task:
313
314 PRINTF("[%s] End of DTLS Client sample\n", __func__);
315
316 //Deinit session
317 mbedtls_net_free(&server_ctx);
318
319 //Deinit SSL context
320 mbedtls_ssl_free(&ssl_ctx);
321
322 //Deinit SSL config
323 mbedtls_ssl_config_free(&ssl_conf);
324
325 //Deinit CTR-DRBG context
326 mbedtls_ctr_drbg_free(&ctr_drbg);
327
328 //Deinit Entropy context
329 mbedtls_entropy_free(&entropy);
330
331 vTaskDelete(NULL);
332
333 return ;
334}
