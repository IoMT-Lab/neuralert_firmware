90static int rnd_pseudo_rand(void *rng_state, unsigned char *output, size_t len)
91{
92 rnd_pseudo_info *info = (rnd_pseudo_info *) rng_state;
93 uint32_t i, *k, sum, delta = 0x9E3779B9;
94 unsigned char result[4], *out = output;
95
96 if (rng_state == NULL) {
97 return (rnd_std_rand(NULL, output, len));
98 }
99
100 k = info->key;
101
102 while (len > 0) {
103 size_t use_len = ( len > 4 ) ? 4 : len;
104 sum = 0;
105
106 for (i = 0; i < 32; i++) {
107 info->v0 += (((info->v1 << 4) ^ (info->v1 >> 5 ) ) + info->v1) ^
108 (sum + k[sum & 3]);
109 sum += delta;
110 info->v1 += (((info->v0 << 4) ^ (info->v0 >> 5)) + info->v0) ^
111 (sum + k[( sum >> 11 ) & 3]);
112 }
113
114 PUT_UINT32_BE(info->v0, result, 0);
115 memcpy(out, result, use_len);
116 len -= use_len;
117 out += 4;
118 }
119
120 return ( 0 );
121}
