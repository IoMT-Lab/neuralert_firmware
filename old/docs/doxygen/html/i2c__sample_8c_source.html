72void i2c_sample(void *param)
73{
74 DA16X_UNUSED_ARG(param);
75
76 PRINTF("I2C_SAMPLE\n");
77 int status;
78 HANDLE I2C;
79
80 status = FALSE;
81
82 // Device Address for AT24C512
83 UINT32 addr = 0xa0;
84
85 // I2C Working Clock [KHz]
86 UINT32 i2c_clock = 400;
87
88 // Buffer for read from EEPROM
89 UINT8 i2c_data_read[AT_I2C_DATA_LENGTH];
90
91 UINT8 i2c_data[AT_I2C_DATA_LENGTH + AT_I2C_LENGTH_FOR_WORD_ADDRESS] = {0,};
92
94 // Board Initialization
96
97 Board_initialization();
98
100 // I2C EEPROM Test
102
103 PRINTF("I2C EEPROM Test start ...\r\n");
104
105 DA16X_CLOCK_SCGATE->Off_DAPB_I2CM = 0;
106 DA16X_CLOCK_SCGATE->Off_DAPB_APBS = 0;
107
108 // Create Handle for I2C Device
109 I2C = DRV_I2C_CREATE(i2c_0);
110
111 // Initialization I2C Device
112 DRV_I2C_INIT(I2C);
113
114 // Set Address for Atmel eeprom AT24C512
115 DRV_I2C_IOCTL(I2C, I2C_SET_CHIPADDR, &addr);
116
117 // Set I2C Working Clock. Unit = KHz
118 DRV_I2C_IOCTL(I2C, I2C_SET_CLOCK, &i2c_clock);
119
120 // GPIO Select for I2C working. GPIO1 = SCL, GPIO0= SDA
121 // _da16x_io_pinmux(PIN_AMUX, AMUX_I2Cm);
122
123 // Data Random Write to EEPROM
124 // Address = 0, Length = 32, Word Address Length = 2
125 // [Start] - [Device addr. W] - [1st word addr.] - [2nd word addr.] - [wdata0] ~ [wdata31] - [Stop]
126
127 i2c_data[0] = AT_I2C_FIRST_WORD_ADDRESS; //Word Address to Write Data. 2 Bytes. refer at24c512 DataSheet
128 i2c_data[1] = AT_I2C_SECOND_WORD_ADDRESS; //Word Address to Write Data. 2 Bytes. refer at24c512 DataSheet
129
130 // Fill Ramp Data
131 for (int i = 0; i < AT_I2C_DATA_LENGTH; i++) {
132 i2c_data[i + AT_I2C_LENGTH_FOR_WORD_ADDRESS] = i;
133 }
134
135 status = DRV_I2C_WRITE(I2C, i2c_data,
136 // Handle, buffer, length, stop enable, dummy
137 AT_I2C_DATA_LENGTH + AT_I2C_LENGTH_FOR_WORD_ADDRESS, 1, 0);
138
139 if (status != TRUE) {
140 PRINTF("ret : 0x%08x\r\n", status);
141 }
142
143 // Delay for AT24C512
144 vTaskDelay(2);
145
146 // Data Random Read from EEPROM
147 // Address = 0, Length = 32, Word Address Length = 2
148 // [Start] - [Device addr. W] - [1st word addr.] - [2nd word addr.] - [Start] -[Device addr. R] - [rdata0] ~ [rdata31] - [Stop]
149
150 // Word Address to Write Data. 2 Bytes. refer at24c512 DataSheet
151 i2c_data_read[0] = AT_I2C_FIRST_WORD_ADDRESS;
152
153 //Word Address to Write Data. 2 Bytes. refer at24c512 DataSheet
154 i2c_data_read[1] = AT_I2C_SECOND_WORD_ADDRESS;
155
156 // Handle, buffer, length, address length, dummy
157 status = DRV_I2C_READ(I2C, i2c_data_read, AT_I2C_DATA_LENGTH, AT_I2C_LENGTH_FOR_WORD_ADDRESS, 0);
158
159 if (status != TRUE) {
160 PRINTF("ret : 0x%08x\r\n", status);
161 }
162
163 // Check Data
164 for (int i = 0; i < AT_I2C_DATA_LENGTH; i++) {
165 if (i2c_data_read[i] != i2c_data[i + AT_I2C_LENGTH_FOR_WORD_ADDRESS]) {
166 PRINTF(" %dth data is different W:0x%02x, R:0x%02x\r\n", i,
167 i2c_data[i + AT_I2C_LENGTH_FOR_WORD_ADDRESS],
168 i2c_data_read[i]);
169 status = AT_I2C_ERROR_DATA_CHECK;
170 }
171 }
172
173 if (status != AT_I2C_ERROR_DATA_CHECK) {
174 PRINTF("***** 32 Bytes Data Write and Read Success *****\r\n");
175 }
176
177 // Data Current Address Read from EEPROM
178 // Length = 32, Word Address Length = 0
179 // [Start] -[Device addr. R] - [rdata0] ~ [rdata31] - [Stop]
180
181 // Handle, buffer, length, address length, dummy
182 status = DRV_I2C_READ(I2C, i2c_data_read, 4, 0, 0);
183
184 if (status != TRUE) {
185 PRINTF("ret : 0x%08x\r\n", status);
186 }
187
188 vTaskDelete(NULL);
189
190 return ;
191}
