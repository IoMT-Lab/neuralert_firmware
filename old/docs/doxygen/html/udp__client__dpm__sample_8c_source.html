165void udp_client_dpm_sample_init_user_config(dpm_user_config_t *user_config)
166{
167 const int session_idx = 0;
168
169 if (!user_config) {
170 PRINTF(" [%s] Invalid parameter\n", __func__);
171 return ;
172 }
173
174 //Set Boot init callback
175 user_config->bootInitCallback = udp_client_dpm_sample_init_callback;
176
177 //Set DPM wakeup init callback
178 user_config->wakeupInitCallback = udp_client_dpm_sample_wakeup_callback;
179
180 //Set External wakeup callback
181 user_config->externWakeupCallback = udp_client_dpm_sample_external_callback;
182
183 //Set Error callback
184 user_config->errorCallback = udp_client_dpm_sample_error_callback;
185
186 //Set session type(UDP Client)
187 user_config->sessionConfig[session_idx].sessionType = REG_TYPE_UDP_CLIENT;
188
189 //Set local port
190 user_config->sessionConfig[session_idx].sessionMyPort =
191 UDP_CLIENT_DPM_SAMPLE_DEF_CLIENT_PORT;
192
193 //Set server IP address
194 memcpy(user_config->sessionConfig[session_idx].sessionServerIp,
195 srv_info.ip_addr, strlen(srv_info.ip_addr));
196
197 //Set server port
198 user_config->sessionConfig[session_idx].sessionServerPort = srv_info.port;
199
200 //Set Connection callback
201 user_config->sessionConfig[session_idx].sessionConnectCallback =
202 udp_client_dpm_sample_connect_callback;
203
204 //Set Recv callback
205 user_config->sessionConfig[session_idx].sessionRecvCallback =
206 udp_client_dpm_sample_recv_callback;
207
208 //Set user configuration
209 user_config->ptrDataFromRetentionMemory = (UCHAR *)&srv_info;
210 user_config->sizeOfRetentionMemory = sizeof(udp_client_dpm_sample_svr_info_t);
211
212 return ;

215unsigned char udp_client_dpm_sample_load_server_info()
216{
217 char *srv_ip_str = NULL;
218 int srv_port = 0;
219
220 udp_client_dpm_sample_svr_info_t *srv_info_ptr = &srv_info;
221
222 memset(srv_info_ptr, 0x00, sizeof(udp_client_dpm_sample_svr_info_t));
223
224 //Read server IP address in nvram
225 srv_ip_str = read_nvram_string(UDP_CLIENT_DPM_SAMPLE_NVRAM_SERVER_IP_NAME);
226 if (srv_ip_str == NULL) {
227 srv_ip_str = UDP_CLIENT_DPM_SAMPLE_DEF_SERVER_IP;
228 }
229
230 //Set server IP address
231 strcpy(srv_info_ptr->ip_addr, srv_ip_str);
232
233 //Read server port in nvram
234 if (read_nvram_int(UDP_CLIENT_DPM_SAMPLE_NVRAM_SERVER_PORT_NAME, &srv_port)) {
235 srv_port = UDP_CLIENT_DPM_SAMPLE_DEF_SERVER_PORT;
236 }
237
238 //Set server port
239 srv_info_ptr->port = srv_port;
240
241 PRINTF(" [%s] UDP Server Information(%s:%d)\n",
242 __func__, srv_info_ptr->ip_addr, srv_info_ptr->port);
243
244 return pdPASS;
245}

247void udp_client_dpm_sample(ULONG arg)
248{
249 DA16X_UNUSED_ARG(arg);
250
251 unsigned char status = pdPASS;
252
253 PRINTF(" [%s] Start UDP Client Sample\n", __func__);
254
255 //Load server information
256 status = udp_client_dpm_sample_load_server_info();
257 if (status == pdFAIL) {
258 PRINTF(" [%s] Failed to load UDP Server information(0x%02x)\n",
259 __func__, status);
260 vTaskDelete(NULL);
261 return ;
262 }
263
264 //Register user configuration
265 dpm_mng_regist_config_cb(udp_client_dpm_sample_init_user_config);
266
267 //Start UDP Client Sample
268 dpm_mng_start();
269
270 vTaskDelete(NULL);
271
272 return ;
273}
