82static int get_data_from_uart1(char *buf)
83{
84 int ch = 0;
85 int i = 0;
86
87 while (1) {
88 /* Get on byte from uart1 com port */
89 ch = getchar_UART(sample_uart_idx, portMAX_DELAY);
90
91 if (ch < 0) {
92 PRINTF("UART error occur\n");
93 vTaskDelay(1);
94 continue;
95 }
96
97 if (echo_enable == TRUE) {
98 puts_UART(sample_uart_idx, (char *)&ch, sizeof(char)); // echo
99 }
100
101 /* check data length */
102 if (i >= (USER_UART1_BUF_SZ - 1)) {
103 i = USER_UART1_BUF_SZ - 2;
104 }
105
106 if (ch == USER_DELIMITER_1 || ch == USER_DELIMITER_2) {
107 buf[i] = USER_DELIMITER_0;
108 break;
109 } else {
110 buf[i++] = ch;
111 }
112 }
113 return (i);
114}

191void run_uart1_sample(UINT32 arg)
192{
193 DA16X_UNUSED_ARG(arg);
194
195 int status;
196
197 /* pinmux config for UART1 - GPIOA[5:4] */
198 _da16x_io_pinmux(PIN_CMUX, CMUX_UART1d);
199
200 /*
201 *
202 ****************************************************************************************
203 * @brief Set UART configuration
204 * @param[in] uart_idx Index value of UART interface (UART_UNIT_1, UART_UNIT_2)
205 * @param[in] *uart_conf_info Buffer pointer of configuration data
206 * @param[in] atcmd_flag AT-CMD flag
207 * @return TX_SUCCESS if succeed to process, others else
208 ****************************************************************************************
209 *
210 * int set_user_UART_conf(int uart_idx, uart_info_t *uart_conf_info, char atcmd_flag)
211 */
212 status = set_user_UART_conf(UART_UNIT_1, &sample_UART_config_info, FALSE);
213 if (status != 0) {
214 PRINTF("[%S] Error to configure for UART1 !!!\n", __func__);
215 return;
216 }
217
218
228 status = UART_init(sample_uart_idx);
229 if (status != 0) {
230 PRINTF("[%S] Error to initialize UART1 with sample_UART_config !!!\n", __func__);
231 return;
232 }
233
234
240#ifdef UART_ERR_CALLBACK
241 UART_IOCTL(get_UART_handler(sample_uart_idx), UART_SET_OVERRUN_INT_CALLBACK, (void *)UART_overrun_err_callback);
242
243 /* Below error interrupts can be tested under parity enabled. */
244 UART_IOCTL(get_UART_handler(sample_uart_idx), UART_SET_PARITY_INT_CALLBACK, (void *)UART_parity_err_callback);
245 UART_IOCTL(get_UART_handler(sample_uart_idx), UART_SET_ERR_INT_CALLBACK, (void *)UART_err_int_callback);
246 UART_IOCTL(get_UART_handler(sample_uart_idx), UART_SET_FRAME_INT_CALLBACK, (void *)UART_frame_err_callback);
247 UART_IOCTL(get_UART_handler(sample_uart_idx), UART_SET_BREAK_INT_CALLBACK, (void *)UART_break_err_callback);
248#endif
249
256#ifdef UART_DATA_FLUSH
257 UART_FLUSH(get_UART_handler(sample_uart_idx));
258#endif
259
260 /* Start UART monitor */
261 uart1_sample();
262}
