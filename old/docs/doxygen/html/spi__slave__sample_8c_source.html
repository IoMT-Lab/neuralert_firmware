146 xTicksToWait )

195void sample_process_general_command_spi()
196{
197 unsigned int data;
198 st_host_request *host_req;
199 st_host_response host_resp;
200
201 data = DA16X_SLAVECTRL->Cmd_Addr;;
202
203 if (data == HOST_ILLEGAL_CMD) {
204 PRINT_HOST("!!!!! Illegal Status!!!!\r\n");
205 return;
206 }
207
208 host_req = (st_host_request *)(&data);
209 DA16X_SLAVECTRL->Cmd_Addr = HOST_ILLEGAL_CMD; // Clear Command Register
210
211
212 // PRINT_HOST("SLAVE_INT_STATUS_CMD length = %d, cmd 0x%02x\r\n", host_req->host_write_length, host_req->host_cmd);
213 switch (host_req->host_cmd) {
214 case HOST_MEM_WRITE_REQ:
215 host_data_buffer = host_data_buffer_allocated;
216 state_slave_proc = STATE_WAIT_G_WR_RES;
217
218 state_slave_proc_next = STATE_WAIT_G_WR_DATA;
219
220 if (host_data_buffer != NULL) {
221 host_data_len_to_write = htons(host_req->host_write_length);
222 host_response_sample_spi((unsigned int)host_data_buffer,
223 (unsigned int)host_req->host_write_length,
224 host_req->host_cmd + 1,
225 1);
226 } else {
227 host_response_sample_spi(0xffffffff,
228 0,
229 HOST_MEM_WRITE_RES,
230 1);
231 }
232
233 break;
234
235 case HOST_MEM_READ_REQ:
236 xSemaphoreTake(host_if_lock, (TickType_t) portMAX_DELAY);
237
238 if (host_data_buffer != NULL) {
239 host_resp.buf_address = (unsigned int)host_data_buffer;
240 host_resp.host_length = host_req->host_write_length;
241 host_resp.resp = HOST_MEM_READ_RES;
242 host_resp.dummy = 0;
243 DA16X_SLAVECTRL->Base_Addr = ((unsigned int)(host_data_buffer)) >>
244 16; //Set Addr.High to Base Address Register
245
246 } else {
247 host_resp.buf_address = 0xffffffff;
248 host_resp.host_length = 0;
249 host_resp.resp = HOST_MEM_READ_RES;
250 host_resp.dummy = 0;
251 }
252
253 DA16X_SLAVECTRL->Resp_Addr1 = *((unsigned int *)(&host_resp));
254 DA16X_SLAVECTRL->Resp_Addr2 = *(((unsigned int *)(&host_resp)) + 1);
255 DA16X_DIOCFG->EXT_INTB_SET = 0x1; // Trigger Host Int (GPIO )
256
257 state_slave_proc = STATE_WAIT_G_RD_RES;
258 state_slave_proc_next = STATE_WAIT_G_RD_DATA;
259
260 xSemaphoreGive(host_if_lock);
261 break;
262
263 }
264}

268void sample_process_data_spi()
269{
270 switch (state_slave_proc) {
271 case STATE_WAIT_G_WR_RES:
272 state_slave_proc = state_slave_proc_next;
273
274 if (state_slave_proc_next == STATE_WAIT_G_WR_DATA) {
275 DA16X_SLAVECTRL->Base_Addr = ((unsigned int)(host_data_buffer)) >>
276 16; //Set Addr.High to Base Address Register
277 }
278
279 break;
280
281 case STATE_WAIT_G_WR_DATA:
282 if (_host_rx_callback != NULL) {
283 _host_rx_callback((unsigned char *)host_data_buffer_allocated, host_data_len_to_write);
284
285 }
286
287 break;
288
289 case STATE_WAIT_G_RD_RES:
290 state_slave_proc = state_slave_proc_next;
291
292 if (state_slave_proc_next == STATE_WAIT_G_RD_DATA)
293 /* Set Addr.High to Base Address Register */
294 {
295 DA16X_SLAVECTRL->Base_Addr = ((unsigned int)(host_data_buffer)) >> 16;
296 }
297
298 // PRINT_HOST("STATE_WAIT_G_RD_RES\r\n");
299 break;
300
301 case STATE_WAIT_G_RD_DATA:
302#if 1 // for RW loop test
303 if (host_data_buffer != NULL) {
304 host_data_buffer = NULL;
305 // PRINT_HOST("Free Allocated Buffer\r\n");
306 }
307
308#endif
309 break;
310
311 default:
312 break;
313
314 }

317void sample_process_command_spi(void)
318{
319 ULONG mask_evt ;
320
321 //Wait SPI Slave Interrupt
322
323 mask_evt = xEventGroupWaitBits(slave_rx_int_evt, slave_int_mask, pdTRUE, pdFALSE, 20);
324 if (mask_evt & SLAVE_INT_STATUS_CMD ) {
325 return;
326 } else {
327 Printf("process command time out\r\n");
328 }
329
330
331 // Clear slave_rx_int_evt <= Driver Event Set
332 xEventGroupClearBits(slave_rx_int_evt, mask_evt );
333
334 //If Tx from Slave Stage , ignore interrupt
335 if (slave_tx_flag) {
336 return;
337 }
338
339 //Command Parser
340 if (mask_evt & SLAVE_INT_STATUS_CMD) {
341 sample_process_general_command_spi();
342 }
343
344 slave_interrupt_acc++;
345}

347void sample_interrupt_handler_spi(void)
348{
349 BaseType_t xHigherPriorityTaskWoken, xResult;
350 xResult = xEventGroupSetBitsFromISR(slave_rx_int_evt, (DA16X_SLAVECTRL->SPI_Intr_Status), &xHigherPriorityTaskWoken);
351
352 if (xResult != pdFAIL ) {
353 /*
354 * If xHigherPriorityTaskWoken is now set to pdTRUE then a context
355 * switch should be requested. The macro used is port specific and will
356 * be either portYIELD_FROM_ISR() or portEND_SWITCHING_ISR() - refer to
357 * the documentation page for the port being used.
358 */
359 portYIELD_FROM_ISR( xHigherPriorityTaskWoken );
360 }
361 // xEventGroupSetBits(adc->mutex, (ADC_DMA_HISR_DONE));
362
363 FPGA_DBG_TIGGER(DA16X_SLAVECTRL->SPI_Intr_Status);
364 DA16X_SLAVECTRL->SPI_Intr_Status |= ( SLAVE_INT_STATUS_ATCMD | SLAVE_INT_STATUS_CMD |
365 SLAVE_INT_STATUS_END); //mask all
366}

375void run_spi_slave_sample(ULONG arg)
376{
377 DA16X_UNUSED_ARG(arg);
378
379 //Clock Gating Off
380 DA16X_CLOCK_SCGATE->Off_SSI_SPI = 0;
381
382 //Initialize SPI Slave HW
383 if (hostif == HOSTIF_SPI) {
384 host_spi_slave_init();
385 }
386
387 vTaskDelay(10);
388 _da16x_io_pinmux(PIN_UMUX, UMUX_GPIO);
389 _da16x_io_pinmux(PIN_AMUX, AMUX_SPIs);
390 _da16x_io_pinmux(PIN_BMUX, BMUX_SPIs);
391
392 //Set GPIOC6 as Interrupt Pin
393 static HANDLE gpio;
394 gpio = GPIO_CREATE(GPIO_UNIT_C);
395 GPIO_INIT(gpio);
396
397 PRINT_HOST("gpio num: %x\n", GPIO_ALT_FUNC_GPIO6);
398 GPIO_SET_ALT_FUNC(gpio, GPIO_ALT_FUNC_EXT_INTR,
399 (GPIO_ALT_GPIO_NUM_TYPE)(GPIO_ALT_FUNC_GPIO6));
400 GPIO_CLOSE(gpio);
401
402 host_data_buffer_allocated = (unsigned int *)
403 sample_tx_buf; //(unsigned int*)DRIVER_MALLOC(4096);
404 PRINTF("BUF Address = 0x%x\r\n", (unsigned int)host_data_buffer_allocated);
405
406 //Interrupt Register
407 _sys_nvic_write(SPI_Slave_IRQn, (void *)sample_slave_interrupt_spi, 0x7);
408
409 //Enable Memory Protection except Buffer and Required Register
410#ifdef SRAM_PROTECT
411 host_sram_protect(host_data_buffer_allocated, 4096, (UINT32 *)0x50080200, 0x7c);
412#endif
413
414 //Create Semaphore for Lock SPI Protocol
415 host_if_lock = xSemaphoreCreateMutex();
416 if (host_if_lock == NULL) {
417 PRINTF("host if lock semaphore NG!!!\n");
418 }
419
420 //Create event for Interrupt Service Routine
421 slave_rx_int_evt = xEventGroupCreate();
422 if (slave_rx_int_evt == NULL) {
423 PRINTF("[%s] slave_rx_int_evt Event Flags Create Error!\n", __func__);
424 }
425
426 //Create SPI Slave Thread
427 BaseType_t xResult;
428
429 xResult = xTaskCreate( sample_thread_spi, "@Host_drv", host_drv_thread_stacksize, ( void * ) NULL, 30, &host_drv_thread_type );
430 if (xResult != pdPASS) {
431 PRINTF("[%s] Failed to created Host_drv create fail(%d) \r\n", __func__, xResult);
432 }
433
434 vTaskDelete(NULL);
435}
