56int crypto_sample_pkcs5()
57{
58 mbedtls_md_context_t sha1_ctx;
59 const mbedtls_md_info_t *info_sha1;
60 unsigned char *key = NULL;
61 int ret = -1;
62
63 key = crypto_sample_calloc(64, sizeof(unsigned char));
64 if (!key) {
65 PRINTF("[%s] Failed to allocate key buffer\n", __func__);
66 goto cleanup;
67 }
68
69 // Initialize a SHA-1 context.
70 mbedtls_md_init(&sha1_ctx);
71
72 // Get the message-digest information associated with the given digest type.
73 info_sha1 = mbedtls_md_info_from_type(MBEDTLS_MD_SHA1);
74 if (info_sha1 == NULL) {
75 PRINTF("[%s] Unknown Hash Type\n", __func__);
76 goto cleanup;
77 }
78
79 // Select the message digest algorithm to use, and allocates internal structures.
80 ret = mbedtls_md_setup(&sha1_ctx, info_sha1, 1);
81 if (ret) {
82 PRINTF("[%s] Failed to setup the message digest algorithm(0x%0x)\n",
83 __func__, -ret);
84 goto cleanup;
85 }
86
87 PRINTF("* PBKDF2 (SHA1): ");
88
89 // Derive a key from a password using PBKDF2 function with HMAC
90 ret = mbedtls_pkcs5_pbkdf2_hmac(&sha1_ctx,
91 pkcs5_password, pkcs5_plen,
92 pkcs5_salt, pkcs5_slen,
93 pkcs5_it_cnt,
94 pkcs5_key_len, key);
95 if (ret) {
96 PRINTF("failed - [%s] Failed to verify key(0x%x)\n", __func__, -ret);
97 goto cleanup;
98 }
99
100 // Check result.
101 if ((ret = memcmp(pkcs5_result_key, key, pkcs5_key_len)) != 0) {
102 PRINTF("failed - [%s] Not matched result", __func__);
103 goto cleanup;
104 }
105
106 PRINTF("passed\n");
107
108cleanup:
109
110 /* Clear the internal structure of ctx and frees any embedded internal structure,
111 * but does not free ctx itself.
112 */
113 mbedtls_md_free(&sha1_ctx);
114
115 if (key) {
116 crypto_sample_free(key);
117 }
118
119 return ret;
120}
