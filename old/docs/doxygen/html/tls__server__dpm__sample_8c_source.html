224void tls_server_dpm_sample_secure_callback(void *config)
225{
226 int ret = 0;
227
228 SECURE_INFO_T *secure_config = (SECURE_INFO_T *)config;
229
230 ret = mbedtls_ssl_config_defaults(secure_config->ssl_conf,
231 MBEDTLS_SSL_IS_SERVER,
232 MBEDTLS_SSL_TRANSPORT_STREAM,
233 MBEDTLS_SSL_PRESET_DEFAULT);
234 if (ret) {
235 PRINTF(" [%s] Failed to set default ssl config(0x%x)\n", __func__, -ret);
236 return ;
237 }
238
239 //import test certificate
240 ret = mbedtls_x509_crt_parse(secure_config->cert_crt,
241 tls_server_dpm_sample_cert,
242 tls_server_dpm_sample_cert_len);
243 if (ret) {
244 PRINTF("[%s] Failed to parse cert(0x%x)\n", __func__, -ret);
245 return ;
246 }
247
248 ret = mbedtls_pk_parse_key(secure_config->pkey_ctx,
249 tls_server_dpm_sample_key,
250 tls_server_dpm_sample_key_len,
251 NULL, 0);
252 if (ret) {
253 PRINTF("[%s] Failed to parse private key(0x%x)\n", __func__, -ret);
254 return ;
255 }
256
257 if (mbedtls_pk_get_type(secure_config->pkey_ctx) == MBEDTLS_PK_RSA) {
258 ret = mbedtls_pk_setup_rsa_alt(secure_config->pkey_alt_ctx,
259 (void *)mbedtls_pk_rsa(*secure_config->pkey_ctx),
260 tls_server_dpm_sample_rsa_decrypt_func,
261 tls_server_dpm_sample_rsa_sign_func,
262 tls_server_dpm_sample_rsa_key_len_func);
263 if (ret) {
264 PRINTF("[%s] Failed to set rsa alt(0x%x)\n", __func__, -ret);
265 return ;
266 }
267
268 ret = mbedtls_ssl_conf_own_cert(secure_config->ssl_conf,
269 secure_config->cert_crt,
270 secure_config->pkey_alt_ctx);
271 if (ret) {
272 PRINTF("[%s] Failed to set cert & private key(0x%x)\n", __func__, -ret);
273 return ;
274 }
275 } else {
276 ret = mbedtls_ssl_conf_own_cert(secure_config->ssl_conf,
277 secure_config->cert_crt,
278 secure_config->pkey_ctx);
279 if (ret) {
280 PRINTF("[%s] Failed to set cert & private key(0x%x)\n", __func__, -ret);
281 return ;
282 }
283 }
284
285 ret = dpm_mng_setup_rng(secure_config->ssl_conf);
286 if (ret) {
287 PRINTF(" [%s:%d] Failed to set random generator(0x%x\n", __func__, __LINE__, -ret);
288 return;
289 }
290
291 //Don't care verification in this sample.
292 mbedtls_ssl_conf_authmode(secure_config->ssl_conf, MBEDTLS_SSL_VERIFY_NONE);
293
294 ret = mbedtls_ssl_setup(secure_config->ssl_ctx, secure_config->ssl_conf);
295 if (ret) {
296 PRINTF("[%s] Failed to set ssl config(0x%x)\n", __func__, -ret);
297 return ;
298 }
299
300 dpm_mng_job_done(); //Done opertaion
301
302 return ;
303}

305void tls_server_dpm_sample_init_user_config(dpm_user_config_t *user_config)
306{
307 const int session_idx = 0;
308
309 if (!user_config) {
310 PRINTF(" [%s] Invalid parameter\n", __func__);
311 return ;
312 }
313
314 //Set Boot init callback
315 user_config->bootInitCallback = tls_server_dpm_sample_init_callback;
316
317 //Set DPM wakkup init callback
318 user_config->wakeupInitCallback = tls_server_dpm_sample_wakeup_callback;
319
320 //Set Error callback
321 user_config->errorCallback = tls_server_dpm_sample_error_callback;
322
323 //Set session type(TCP Server)
324 user_config->sessionConfig[session_idx].sessionType = REG_TYPE_TCP_SERVER;
325
326 //Set local port
327 user_config->sessionConfig[session_idx].sessionMyPort =
328 TLS_SERVER_DPM_SAMPLE_DEF_SERVER_PORT;
329
330 //Set Connection callback
331 user_config->sessionConfig[session_idx].sessionConnectCallback =
332 tls_server_dpm_sample_connect_callback;
333
334 //Set Recv callback
335 user_config->sessionConfig[session_idx].sessionRecvCallback =
336 tls_server_dpm_sample_recv_callback;
337
338 //Set secure mode
339 user_config->sessionConfig[session_idx].supportSecure = pdTRUE;
340
341 //Set secure setup callback
342 user_config->sessionConfig[session_idx].sessionSetupSecureCallback =
343 tls_server_dpm_sample_secure_callback;
344
345 //Set user configuration
346 user_config->ptrDataFromRetentionMemory = (UCHAR *)&srv_info;
347 user_config->sizeOfRetentionMemory = sizeof(tls_server_dpm_sample_svr_info_t);
348
349 return ;
350}

352unsigned char tls_server_dpm_sample_load_server_info()
353{
354 int srv_port = 0;
355
356 tls_server_dpm_sample_svr_info_t *srv_info_ptr = &srv_info;
357
358 memset(srv_info_ptr, 0x00, sizeof(tls_server_dpm_sample_svr_info_t));
359
360 //Read server port in nvram
361 if (read_nvram_int(TLS_SERVER_DPM_SAMPLE_NVRAM_SERVER_PORT_NAME, &srv_port)) {
362 srv_port = TLS_SERVER_DPM_SAMPLE_DEF_SERVER_PORT;
363 }
364
365 //Set server port
366 srv_info_ptr->port = srv_port;
367
368 PRINTF(" [%s] TLS Server Information(%d)\n", __func__, srv_info_ptr->port);
369
370 return pdPASS;
371}

373void tls_server_dpm_sample(void *param)
374{
375 DA16X_UNUSED_ARG(param);

395
396 vTaskDelete(NULL);
397
398 return ;
