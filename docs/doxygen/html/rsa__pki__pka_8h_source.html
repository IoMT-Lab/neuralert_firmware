330 /* Add: Res = OpA + OpB */
331 #define RSA_PKA_Add(LenID, OpA, OpB, Res, Tag, VirtualHwBaseAddr) \
332  _RSA_PKA_ExecOperation( PkaAdd,(LenID), 0, (OpA), 0, (OpB), 0, (Res), (Tag), (VirtualHwBaseAddr) )
333 /* AddIm: Res = OpA + OpBIm */
334 #define RSA_PKA_Add_IM(LenID, OpA, OpBIm, Res, Tag, VirtualHwBaseAddr) \
335  _RSA_PKA_ExecOperation( PkaAddIm,(LenID), 0, (OpA), 1, (OpBIm), 0, (Res), (Tag), (VirtualHwBaseAddr) )
336 /* SubIm: Res = OpA - OpBIm */
337 #define RSA_PKA_SUB_IM(LenID, OpA, OpBIm, Res, Tag, VirtualHwBaseAddr) \
338  _RSA_PKA_ExecOperation( PkaSubIm,(LenID), 0, (OpA), 1, (OpBIm), 0, (Res), (Tag), (VirtualHwBaseAddr) )
339 
340 /*----------------------------------*/
341 /* 2. Logical operations */
342 /*----------------------------------*/
343 
344 /* Clr: Res = OpA & 0 - clears the operand A. */
345 #define RSA_PKA_Clr(LenID, OpA, Tag, VirtualHwBaseAddr) \
346  _RSA_PKA_ExecOperation( PkaAND, (LenID), 0, (OpA), 1, 0x00, 0, (OpA), (Tag), (VirtualHwBaseAddr))
347 
348 /* Clear: for full clearing the actual register opA, this macro calls Clr operation twice. */
349 #define RSA_PKA_Clear(LenID, OpA, Tag, VirtualHwBaseAddr) \

351  RSA_PKA_Clr(LenID, OpA, Tag, VirtualHwBaseAddr)

353 /* Set0: Res = OpA || 1 : set bit0 = 1, other bits are not changed */
