99static UINT user_sample_da16_fw_update(void)
100{
101 UINT status = OTA_SUCCESS;
102
103 PRINTF("\n\n>>> Start sample : DA16200 FW update \n\n\n");
104
105 memset(g_ota_update_conf, 0x00, sizeof(OTA_UPDATE_CONFIG));
106
107 /* Setting the type to be updated */
108 g_ota_update_conf->update_type = OTA_TYPE_RTOS;
109
110 /* url setting example - Change it to suit your environment. */
111 memcpy(g_ota_update_conf->url, ota_server_url_rtos, strlen(ota_server_url_rtos));
112
113 /* If you downloaded the FW successfully, the boot_index will change automatically. */
114 g_ota_update_conf->auto_renew = 0;
115
116 /* Called when the download of each FW is finished. (Success or Fail) */
117 g_ota_update_conf->download_notify = user_sample_da16_fw_download_notify;
118
119 /* Called when the renew(change boot index) is complete (If it succeeds, it will reboot automatically.) */
120 g_ota_update_conf->renew_notify = user_sample_da16_fw_renew_notify;
121
122 /* Flag to confirm callback */
123 da16_rtos_download_notify = UNKNOWN_STATUS;
124 da16_renew_notify = UNKNOWN_STATUS;
125
126 /* Start the fw update process */
127 status = ota_update_start_download(g_ota_update_conf);
128 if (status) {
129 PRINTF("[%s] OTA update start fail (0x%02x)\n", __func__, status);
130 return status;
131 }
132
133 /* Waiting for download notification */
134 while (1) {
135 vTaskDelay( 10 / portTICK_PERIOD_MS );
136 if (da16_rtos_download_notify != UNKNOWN_STATUS) {
137 PRINTF("[%s] Download is complete (RTOS status = 0x%02x)\n",
138 __func__, da16_rtos_download_notify);
139 break;
140 }
141 }
142
143 if (da16_rtos_download_notify != OTA_SUCCESS) {
144 return OTA_FAILED;
145 }
146
147 /* Waiting for renew notification */
148 if (g_ota_update_conf->auto_renew == 1) {
149 while (1) {
150 vTaskDelay( 10 / portTICK_PERIOD_MS );
151 if (da16_renew_notify != UNKNOWN_STATUS) {
152 break;
153 }
154 }
155
156 if (da16_renew_notify != OTA_SUCCESS) {
157 return OTA_FAILED;
158 }
159 }
160
161 return OTA_SUCCESS;
162}
163

165/* If you set auto_renew to true(g_ota_update_conf->auto_renew = 1), you do not need to call ota_update_start_renew().*/
