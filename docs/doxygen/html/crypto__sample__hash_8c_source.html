45static void hexify(unsigned char *out, const unsigned char *in, int len)
46{
47 unsigned char l, h;
48
49 while (len != 0) {
50 h = *in / 16;
51 l = *in % 16;
52
53 if (h < 10) {
54 *out++ = '0' + h;
55 } else {
56 *out++ = 'a' + h - 10;
57 }
58
59 if (l < 10) {
60 *out++ = '0' + l;
61 } else {
62 *out++ = 'a' + l - 10;
63 }
64
65 ++in;
66 len--;
67 }
68
69 return ;
70}

75static int unhexify(unsigned char *out, const char *in)
76{
77 unsigned char c, c2;
78 int len = strlen(in) / 2;
79
80 while (*in != 0) {
81 c = *in++;
82
83 if (c >= '0' && c <= '9') {
84 c -= '0';
85 } else if (c >= 'a' && c <= 'f') {
86 c -= 'a' - 10;
87 } else if (c >= 'A' && c <= 'F') {
88 c -= 'A' - 10;
89 }
90
91 c2 = *in++;
92
93 if (c2 >= '0' && c2 <= '9') {
94 c2 -= '0';
95 } else if (c2 >= 'a' && c2 <= 'f') {
96 c2 -= 'a' - 10;
97 } else if (c2 >= 'A' && c2 <= 'F') {
98 c2 -= 'A' - 10;
99 }
100
101 *out++ = (c << 4) | c2;
102 }
103
104 return len;
105}

560int crypto_sample_hash_md_wrapper_info(char *md_name, mbedtls_md_type_t md_type,
561 int md_size)
562{
563 int ret = -1;
564 const mbedtls_md_info_t *md_info = NULL;
565 const int *md_type_ptr = NULL;
566 int found = 0;
567
568 PRINTF(">>> %s: ", md_name);
569
570 // Get the message-digest information associated with the given digest type.
571 md_info = mbedtls_md_info_from_type(md_type);
572 if (!md_info) {
573 PRINTF("[%s] Unknown Hash Type(%d)\r\n", __func__, md_type);
574 goto cleanup;
575 }
576
577 // Get the message-digest information associated with the given digest name.
578 if (md_info != mbedtls_md_info_from_string(md_name)) {
579 PRINTF("[%s] Unknown Hash Name(%s)\r\n", md_name);
580 goto cleanup;
581 }
582
583 // Extract the message-digest type from the message-digest information structure.
584 if (mbedtls_md_get_type(md_info) != (mbedtls_md_type_t)md_type) {
585 PRINTF("[%s] Not matched Hash Type\r\n", __func__);
586 goto cleanup;
587 }
588
589 // Extract the message-digest size from the message-digest information structure.
590 if (mbedtls_md_get_size(md_info) != (unsigned char)md_size) {
591 PRINTF("[%s] Not matched Hash Size\r\n", __func__);
592 goto cleanup;
593 }
594
595 // Extract the message-digest name from the message-digest information structure.
596 if (strcmp(mbedtls_md_get_name(md_info), md_name) != 0) {
597 PRINTF("[%s] Not matched Hash Name\r\n", __func__);
598 goto cleanup;
599 }
600
601 // Find the list of digests supported by the generic digest module.
602 for (md_type_ptr = mbedtls_md_list() ; *md_type_ptr != 0 ; md_type_ptr++) {
603 if (*md_type_ptr == md_type) {
604 found = 1;
605 break;
606 }
607 }
608
609 if (!found) {
610 PRINTF("[%s] Not supported Hash Type(%d)\r\n", __func__, md_type);
611 ret = -1;
612 goto cleanup;
613 } else {
614 ret = 0;
615 }
616
617cleanup:
618
619 PRINTF("%s\r\n", (ret == 0) ? "passed" : "failed");
620
621 return ret;
622}

958
959 PRINTF(">>> %s: ", md_name);
960

962 if (!ctx) {

974 if (!output) {
975 PRINTF("[%s] Failed to allocate output buffer\r\n", __func__);
976 goto cleanup;
977 }
978
979 /* Initialize a message-digest context without binding it
980 * to a particular message-digest algorithm.
981 */
982 mbedtls_md_init(ctx);
983

985

987

1009 }

1013 ret = -1;
1014 goto cleanup;
1015 }
1016

1018 ret = mbedtls_md_update(ctx, (const unsigned char *)text_src_string, halfway);

1030 PRINTF("[%s] Failed to update message-digest(0x%x)\r\n",
1031 __func__, -ret);
1032 goto cleanup;
1033 }
1034

1036 ret = mbedtls_md_finish(ctx, output);
1037 if (ret) {
1038 PRINTF("[%s] Failed to finish the digest operation(0x%x)\r\n",
1039 __func__, -ret);
1040 goto cleanup;
1041 }

1044

1047 PRINTF("[%s] Not matched result\r\n", __func__);
1048 goto cleanup;
1049 }
1050
1051cleanup:
1052
1053 /* Clear the internal structure of ctx and frees any embedded internal structure,
1054 * but does not free ctx itself.
1055 */
1056 mbedtls_md_free(ctx);
1057
1058 if (ctx) {
1059 crypto_sample_free(ctx);
1060 }
1061
1062 if (hash_str) {
1063 crypto_sample_free(hash_str);
1064 }
1065
1066 if (output) {
1067 crypto_sample_free(output);

1070 PRINTF("%s\r\n", (ret == 0) ? "passed" : "failed");
