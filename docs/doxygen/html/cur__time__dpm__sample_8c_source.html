69static void display_cur_time(char *timer_name)
70{
71 DA16X_UNUSED_ARG(timer_name);
72
73 __time64_t now;
74 struct tm *ts;
75 char time_buf[80];
76
77 /* get current time */
78 da16x_time64(NULL, &now);
79 ts = (struct tm *)da16x_localtime64(&now);
80
81 /* make time string */
82 da16x_strftime(time_buf, sizeof(time_buf), "%Y.%m.%d %H:%M:%S", ts);
83
84 /* display current time string */

97static unsigned char set_n_start_SNTP(void)
98{
99 unsigned char status = pdPASS;
100
101 /* Check current SNTP running status */
102 if (dpm_mode_is_wakeup() == DPM_WAKEUP) {
103 status = get_sntp_use_from_rtm();
104 } else {
105 status = get_sntp_use();
106 }
107
108 if (status == pdPASS) {
109 long time_zone;
110
111 /* Already SNTP module running, set again time-zone ... */
112 time_zone = get_timezone_from_rtm();
113 da16x_SetTzoff(time_zone);
114
115 return pdPASS;
116 }
117
118 if (dpm_mode_is_wakeup() == NORMAL_BOOT) {
119 /* Config and save SNTP server URI */
120 status = set_sntp_server((unsigned char *)TEST_SNTP_SERVER, 0);
121 if (status != pdPASS) {
122 PRINTF("[%s] Failed to write nvram operation (SNTP server domain)...\n", __func__);
123 status = pdFAIL;
124 goto _exit;
125 }
126
127 /* Config and save SNTP periodic renew time : seconds */
128 status = set_sntp_period(TEST_SNTP_RENEW_PERIOD);
129 if (status != pdPASS) {
130 PRINTF("[%s] Failed to write nvram operation (SNTP renew period)...\n", __func__);
131 status = pdFAIL;
132 goto _exit;
133 }
134
135 /* Config and save SNTP time zone */
136 set_time_zone(TEST_TIME_ZONE);
137 set_timezone_to_rtm(TEST_TIME_ZONE);
138 da16x_SetTzoff(TEST_TIME_ZONE);
139 set_time_zone(TEST_TIME_ZONE);
140
141 /* Config, save, and run SNTP client */
142 if (set_sntp_use(SNTP_ENABLE) != 0) {
143 PRINTF("[%s] Failed to run SNTP...\n", __func__);
144 status = pdFAIL;
145 goto _exit;
146 }
147
148 /* Save config and start SNTP client */
149 set_sntp_use_to_rtm(status);
150 }
151
152_exit :
153 return status;
154}

156/*
157 * Sample code for getting current time
158 */
159void cur_time_dpm_sample(void *param)
160{
161 DA16X_UNUSED_ARG(param);
162
163 unsigned char status;
164
165 /* Check wakeup type : RTC timer wakeup */
166 if (dpm_mode_is_wakeup() == DPM_WAKEUP) {
167 if (dpm_mode_get_wakeup_type() != DPM_RTCTIME_WAKEUP) {
168 /* Set flag to go to DPM sleep 3 */
169 dpm_app_sleep_ready_set(SAMPLE_CUR_TIME_DPM);
170
171 return;
172 }
173 }
174
175 /* Config SNTP client */
176 status = set_n_start_SNTP();
177
178 if (status == pdFAIL) {
179 PRINTF("[%s] Failed to start SNTP client !!!\n", __func__);
180
181 /* Remove dpm register information */
182 dpm_app_unregister(SAMPLE_CUR_TIME_DPM);
183 vTaskDelete(NULL);
184 return;
185 }
186
187 /* Regist periodic RTC Timer : Get current time */
188 if (dpm_mode_is_wakeup() == NORMAL_BOOT) {
189 /* Time delay for stable running SNTP client */
190 vTaskDelay(10);
191
192 status = dpm_timer_create(SAMPLE_CUR_TIME_DPM,
193 "timer1",
194 display_cur_time,
195 CUR_TIME_LOOP_DELAY,
196 CUR_TIME_LOOP_DELAY);
197
198 switch ((int)status) {
199 case DPM_MODE_NOT_ENABLED :
200 case DPM_TIMER_SEC_OVERFLOW :
201 case DPM_TIMER_ALREADY_EXIST:
202 case DPM_TIMER_NAME_ERROR :
203 case DPM_UNSUPPORTED_RTM :
204 case DPM_TIMER_REGISTER_FAIL:
205 case DPM_TIMER_MAX_ERR :
206 PRINTF(">>> Fail to create %s timer (err=%d)\n", SAMPLE_CUR_TIME_DPM, (int)status);
207 // Delay to display above message on console ...
208 vTaskDelay(2);
209
210 break;
211 }
212
213 /* Set flag to go to DPM sleep 3 */
214 dpm_app_sleep_ready_set(SAMPLE_CUR_TIME_DPM);
215 } else {
216 /* Notice initialize done to DPM module */
217 dpm_app_wakeup_done(SAMPLE_CUR_TIME_DPM);
218 }
219
220 while (1) {
221 /* Nothing to do... */
222 vTaskDelay(100);
223 }
224}
