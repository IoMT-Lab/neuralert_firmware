95static int myrand(void *rng_state, unsigned char *output, size_t len)

110int crypto_sample_rsa_pkcs1()
111{
112 int ret = -1;
113 size_t len = 0;
114 mbedtls_rsa_context *rsa = NULL; // The RSA context structure.
115 unsigned char *rsa_plaintext = NULL;
116 unsigned char *rsa_decrypted = NULL;
117 unsigned char *rsa_ciphertext = NULL;
118#if defined(MBEDTLS_SHA1_C)
119 unsigned char *sha1sum = NULL;
120#endif // (MBEDTLS_SHA1_C)
121
122 rsa = (mbedtls_rsa_context *)crypto_sample_calloc(1, sizeof(mbedtls_rsa_context));
123 if (!rsa) {
124 PRINTF("[%s] Failed to allocate RSA context\r\n", __func__);
125 goto cleanup;
126 }
127
128 rsa_plaintext = crypto_sample_calloc(CRYPTO_SAMPLE_RSA_MARGIN(PT_LEN),
129 sizeof(unsigned char));
130 if (!rsa_plaintext) {
131 PRINTF("[%s] Failed to allocate plain text buffer\r\n", __func__);
132 goto cleanup;
133 }
134
135 rsa_decrypted = crypto_sample_calloc(CRYPTO_SAMPLE_RSA_MARGIN(PT_LEN),
136 sizeof(unsigned char));
137 if (!rsa_decrypted) {
138 PRINTF("[%s] Failed to allocate decryption buffer\r\n", __func__);
139 goto cleanup;
140 }
141
142 rsa_ciphertext = crypto_sample_calloc(KEY_LEN, sizeof(unsigned char));
143 if (!rsa_ciphertext) {
144 PRINTF("[%s] Failed to allocate cipher text buffer\r\n", __func__);
145 goto cleanup;
146 }
147
148#if defined(MBEDTLS_SHA1_C)
149 sha1sum = crypto_sample_calloc(20, sizeof(unsigned char));
150 if (!sha1sum) {
151 PRINTF("[%s] Failed to allocate SHA1 buffer\r\n", __func__);
152 goto cleanup;
153 }
154#endif // (MBEDTLS_SHA1_C)
155
156 // Initializes an RSA context.
157 mbedtls_rsa_init(rsa, MBEDTLS_RSA_PKCS_V15, MBEDTLS_MD_NONE);
158
159 // Import pre-defined RSA information.
160 rsa->len = KEY_LEN;
161 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->N), 16, RSA_N));
162 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->E), 16, RSA_E));
163 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->D), 16, RSA_D));
164 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->P), 16, RSA_P));
165 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->Q), 16, RSA_Q));
166 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->DP), 16, RSA_DP));
167 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->DQ), 16, RSA_DQ));
168 MBEDTLS_MPI_CHK(mbedtls_mpi_read_string(&(rsa->QP), 16, RSA_QP));
169
170 PRINTF("* RSA key validation: ");
171
172 // Check if a context contains at least an RSA public key.
173 ret = mbedtls_rsa_check_pubkey(rsa);
174 if (ret) {
175 PRINTF("failed\r\n");
176 goto cleanup;
177 }
178
179 ret = mbedtls_rsa_check_privkey(rsa);
180 if (ret) {
181 PRINTF("failed\r\n");
182 goto cleanup;
183 }
184
185 PRINTF("passed\r\n");
186
187 PRINTF("* PKCS#1 encryption : ");
188
189 memcpy(rsa_plaintext, RSA_PT, PT_LEN);
190
191 // Add the message padding, then performs an RSA operation.
192 ret = mbedtls_rsa_pkcs1_encrypt(rsa, myrand,
193 NULL, MBEDTLS_RSA_PUBLIC, PT_LEN,
194 rsa_plaintext, rsa_ciphertext);
195 if (ret) {
196 PRINTF( "failed\r\n" );
197 goto cleanup;
198 }
199
200 PRINTF("passed\r\n");
201
202 crypto_sample_hex_dump(" + rsa_ciphertext: ", rsa_ciphertext, PT_LEN);
203
204 PRINTF("* PKCS#1 decryption : ");
205
206 // Perform an RSA operation, then removes the message padding.
207 ret = mbedtls_rsa_pkcs1_decrypt(rsa, myrand,
208 NULL, MBEDTLS_RSA_PRIVATE, &len,
209 rsa_ciphertext, rsa_decrypted,
210 (PT_LEN * sizeof(unsigned char)));
211
212 if (ret) {
213 PRINTF("failed\r\n");
214 goto cleanup;
215 }
216
217 // Compare output with expected result.
218 if ((ret = memcmp(rsa_decrypted, rsa_plaintext, len)) != 0) {
219 PRINTF("failed\r\n");
220 crypto_sample_hex_dump(" + rsa_decrypted: ", rsa_decrypted, len );
221 crypto_sample_hex_dump(" + rsa_plaintext: ", rsa_plaintext, len );
222 goto cleanup;
223 }
224
225 PRINTF("passed\r\n");
226
227 crypto_sample_hex_dump(" + rsa_decrypted: ", rsa_decrypted, len );
228 crypto_sample_hex_dump(" + rsa_plaintext: ", rsa_plaintext, len );
229
230#if defined(MBEDTLS_SHA1_C)
231 PRINTF("* PKCS#1 data sign : ");
232
233 mbedtls_sha1_ret(rsa_plaintext, PT_LEN, sha1sum);
234
235 // Perform a private RSA operation to sign a message digest using PKCS#1.
236 ret = mbedtls_rsa_pkcs1_sign(rsa, myrand,
237 NULL, MBEDTLS_RSA_PRIVATE, MBEDTLS_MD_SHA1,
238 0, sha1sum, rsa_ciphertext);
239 if (ret) {
240 PRINTF( "failed\r\n" );
241 goto cleanup;
242 }
243
244 PRINTF("passed\r\n");
245
246 PRINTF("* PKCS#1 sig. verify: ");
247
248 // Perform a public RSA operation and checks the message digest.
249 ret = mbedtls_rsa_pkcs1_verify(rsa, NULL,
250 NULL, MBEDTLS_RSA_PUBLIC, MBEDTLS_MD_SHA1,
251 0, sha1sum, rsa_ciphertext);
252 if (ret) {
253 PRINTF("failed\r\n");
254 goto cleanup;
255 }
256
257 PRINTF("passed\r\n");
258#endif /* MBEDTLS_SHA1_C */
259
260cleanup:
261
262 // Free the components of an RSA key.
263 mbedtls_rsa_free(rsa);
264
265#if defined(MBEDTLS_SHA1_C)
266 if (sha1sum) {
267 crypto_sample_free(sha1sum);
268 }
269#endif // (MBEDTLS_SHA1_C)
270
271 if (rsa_plaintext) {
272 crypto_sample_free(rsa_plaintext);
273 }
274
275 if (rsa_decrypted) {
276 crypto_sample_free(rsa_decrypted);
277 }
278
279 if (rsa_ciphertext) {
280 crypto_sample_free(rsa_ciphertext);
281 }
282
283 if (rsa) {
284 crypto_sample_free(rsa);
285 }
286
287 return ret;
288}
289#endif // (MBEDTLS_PKCS1_V15)
