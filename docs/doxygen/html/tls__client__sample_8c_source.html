76void tls_client_sample(void *param)
77{
78 DA16X_UNUSED_ARG(param);
79
80 const char *pers = "tls_client";
81 int ret = 0;
82 char data_buffer[TLS_CLIENT_SAMPLE_DEF_BUF_SIZE] = {0x00,};
83 size_t len = sizeof(data_buffer);
84 char str_port[8] = {0x00,};
85
86 mbedtls_net_context server_ctx;
87 mbedtls_ssl_context ssl_ctx;
88 mbedtls_ssl_config ssl_conf;
89 mbedtls_ctr_drbg_context ctr_drbg;
90 mbedtls_entropy_context entropy;
91
92 PRINTF("[%s] Start of TLS Client sample\r\n", __func__);
93
94 //Init session
95 mbedtls_net_init(&server_ctx);
96
97 //Init SSL context
98 mbedtls_ssl_init(&ssl_ctx);
99
100 //Init SSL config
101 mbedtls_ssl_config_init(&ssl_conf);
102
103 //Init CTR-DRBG context
104 mbedtls_ctr_drbg_init(&ctr_drbg);
105
106 //Init Entropy context
107 mbedtls_entropy_init(&entropy);
108
109 PRINTF("\r\nConnecting to tcp/%s:%d...",
110 TLS_CLIENT_SAMPLE_DEF_SERVER_IP_ADDR,
111 TLS_CLIENT_SAMPLE_DEF_SERVER_PORT);
112
113 snprintf(str_port, sizeof(str_port), "%d", TLS_CLIENT_SAMPLE_DEF_SERVER_PORT);
114
115 ret = mbedtls_net_connect(&server_ctx,
116 TLS_CLIENT_SAMPLE_DEF_SERVER_IP_ADDR, str_port,
117 MBEDTLS_NET_PROTO_TCP);
118 if (ret) {
119 PRINTF("\r\n[%s] Failed to connect TLS session(0x%x)\r\n", __func__, -ret);
120 goto end_of_task;
121 }
122
123 PRINTF("ok\r\n");
124
125 PRINTF("Setting up the TLS structure...");
126
127 //Set default configuration
128 ret = mbedtls_ssl_config_defaults(&ssl_conf,
129 MBEDTLS_SSL_IS_CLIENT,
130 MBEDTLS_SSL_TRANSPORT_STREAM,
131 MBEDTLS_SSL_PRESET_DEFAULT);
132 if (ret) {
133 PRINTF("\r\n[%s] Failed to init tls config(0x%x)\r\n", __func__, -ret);
134 goto end_of_task;
135 }
136
137 ret = mbedtls_ctr_drbg_seed(&ctr_drbg, mbedtls_entropy_func, &entropy,
138 (const unsigned char *)pers, strlen(pers));
139 if (ret) {
140 PRINTF("\r\n[%s] Failed to set CTR-DRBG(0x%x)\r\n", __func__, -ret);
141 goto end_of_task;
142 }
143
144 mbedtls_ssl_conf_rng(&ssl_conf, mbedtls_ctr_drbg_random, &ctr_drbg);
145
146 //Don't care verification in this sample.
147 mbedtls_ssl_conf_authmode(&ssl_conf, MBEDTLS_SSL_VERIFY_NONE);
148
149 //Setup an SSL context for use.
150 ret = mbedtls_ssl_setup(&ssl_ctx, &ssl_conf);
151 if (ret) {
152 PRINTF("\r\n[%s] Failed to setup dtls contex(0x%x)\r\n", __func__, -ret);
153 goto end_of_task;
154 }
155
156 mbedtls_ssl_set_bio(&ssl_ctx, &server_ctx,
157 mbedtls_net_send, mbedtls_net_recv, NULL);
158
159 PRINTF("ok\r\n");
160
161 PRINTF("Performing the TLS handshake...");
162
163 while ((ret = mbedtls_ssl_handshake(&ssl_ctx)) != 0) {
164 if (ret == MBEDTLS_ERR_NET_CONN_RESET) {
165 PRINTF("\r\n[%s] Peer closed the connection(0x%x)\r\n", __func__, -ret);
166 goto end_of_task;
167 }
168
169 if ((ret != MBEDTLS_ERR_SSL_WANT_READ)
170 && (ret != MBEDTLS_ERR_SSL_WANT_WRITE)) {
171 PRINTF("\r\n[%s] Failed to do tls handshake(0x%x)\r\n", __func__, -ret);
172 goto end_of_task;
173 }
174 }
175
176 PRINTF("ok\r\n");
177
178 //data transmit
179 do {
180 len = sizeof(data_buffer) - 1;
181 memset(data_buffer, 0x00, sizeof(data_buffer));
182
183 PRINTF("< Read from server: ");
184
185 //Read at most 'len' application data bytes.
186 ret = mbedtls_ssl_read(&ssl_ctx, (unsigned char *)data_buffer, len);
187 if (ret <= 0) {
188 switch (ret) {
189 case MBEDTLS_ERR_SSL_WANT_READ:
190 case MBEDTLS_ERR_SSL_WANT_WRITE:
191 PRINTF("\r\nNeed more data - mbedtls_ssl_read(0x%x)\r\n", -ret);
192 continue;
193 case MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY:
194 PRINTF("\r\nConnection was closed gracefully\r\n");
195 goto end_of_task;
196 case MBEDTLS_ERR_NET_CONN_RESET:
197 PRINTF("\r\nConnection was reset by peer\r\n");
198 goto end_of_task;
199 default:
200 PRINTF("\r\nFailed to read data(0x%x)\r\n", -ret);
201 break;
202 }
203 goto end_of_task;
204 }
205
206 len = ret;
207
208 PRINTF("%d bytes read\r\n", len);
209
210 tls_client_sample_hex_dump("Received data", (unsigned char *)data_buffer, len);
211
212 PRINTF("> Write to server: ");
213
214 //Try to write exactly 'len' application data bytes.
215 while ((ret = mbedtls_ssl_write(&ssl_ctx, (unsigned char *)data_buffer, len)) <= 0) {
216 switch (ret) {
217 case MBEDTLS_ERR_SSL_WANT_READ:
218 case MBEDTLS_ERR_SSL_WANT_WRITE:
219 PRINTF("\r\nNeed more data - mbedtls_ssl_write(0x%x)\r\n", -ret);
220 continue;
221 case MBEDTLS_ERR_SSL_PEER_CLOSE_NOTIFY:
222 PRINTF("\r\nConnection was closed gracefully\r\n");
223 break;
224 case MBEDTLS_ERR_NET_CONN_RESET:
225 PRINTF("\r\nConnection was reset by peer\r\n");
226 break;
227 default:
228 PRINTF("\r\nFailed to write data(0x%x)\r\n", -ret);
229 break;
230 }
231 goto end_of_task;
232 }
233
234 PRINTF("%d bytes written\r\n", len);
235
236 tls_client_sample_hex_dump("Sent data", (unsigned char *)data_buffer, len);
237 } while (1);
238
239end_of_task:
240
241 PRINTF("[%s] End of TLS Client sample\r\n", __func__);
242
243 //Deinit session
244 mbedtls_net_free(&server_ctx);
245
246 //Deinit SSL context
247 mbedtls_ssl_free(&ssl_ctx);
248
249 //Deinit SSL config

251

254
255 //Deinit Entropy context

258 vTaskDelete(NULL);
259
260 return ;
261}
