117  PKA_EXEC_OPERATION( PKA_OPCODE_ID_AND, (lenId), 0, (OpA), 1, 0x1E, 0, (Res), 0 )

181 /* RMul: Res = LowHalfOf(OpA * OpB), where size of operands and result is equaled to operation
182  size, defined by lenId. Note: for receiving full result, the lenId must be set according
183  to (sizeA + sizeB) and leading not significant bits of operands must be zeroed */
184 #define PKA_MUL_LOW(lenId, Res, OpA, OpB) \
185  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MULLOW, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
186 /* HMul: Res = HighHalfOf(OpA * OpB) + one high word of low half of (OpA * OpB), where size of
187  operands is equaled to operation size, defined by lenId. Note: Size of operation result
188  is by one word large, than operation size */
189 #define PKA_MUL_HIGH(lenId, Res, OpA, OpB) \
190  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MULHIGH, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
191 /* ModMul: Res = OpA * OpB mod N - modular multiplication */
192 #define PKA_MOD_MUL(lenId, Res, OpA, OpB) \
193  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODMUL, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
194 /* ModMulN: Res = OpA * OpB mod N - modular multiplication (final reduction is omitted)*
195 * up to PKA_EXTRA_BITS extra bits */
196 #define PKA_MOD_MUL_NFR(lenId, Res, OpA, OpB) \
197  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODMULN, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
198 /* ModMulAcc: Res = OpA * OpB + OpC mod N - modular multiplication and *
199 * adding, result reduced */
200 #define PKA_MOD_MUL_ACC(lenId, Res, OpA, OpB, OpC) \
201  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODMLAC, (lenId), 0, (OpA), 0, (OpB), 0, (Res), (OpC) )
202 /* ModMulAccN: Res = OpA * OpB + OpC mod N - modular multiplication and *
203 * acdding (final reduction is omitted) - up to PKA_EXTRA_BITS extra bits */
204 #define PKA_MOD_MUL_ACC_NFR(lenId, Res, OpA, OpB, OpC) \
205  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODMLACNR, (lenId), 0, (OpA), 0, (OpB), 0, (Res), (OpC) )
206 /* ModExp: Res = OpA ** OpB mod N - modular exponentiation */
207 #define PKA_MOD_EXP(lenId, Res, OpA, OpB) \
208  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODEXP, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
209 /* Divide: Res = OpA / OpB , OpA = OpA mod OpB - division, */
210 #define PKA_DIV(lenId, Res, OpA, OpB) \
211  PKA_EXEC_OPERATION( PKA_OPCODE_ID_DIVISION, (lenId), 0, (OpA), 0, (OpB), 0, (Res), 0 )
212 /* ModInv: Modular inversion: calculates Res = 1/OpB mod N */
213 #define PKA_MOD_INV(lenId, Res, OpB) \
214  PKA_EXEC_OPERATION( PKA_OPCODE_ID_MODINV, (lenId), 1,1, 0,(OpB), 0,(Res), 0 )
215 /* Modular reduction: Res = OpB mod B by subtract the modulus B *
216 * times, while Res > B. Counter C should be set in the Tag bits of Status */
217 #define PKA_REDUCE(lenId, Res, OpA) \
218  PKA_EXEC_OPERATION( PKA_OPCODE_ID_REDUCTION, (lenId), 0, (OpA), 0, 0/*opB not need*/, 0, (Res), 0/*Tag*/ )
