35 * Task control block. A task control block (TCB) is allocated for each task,
36 * and stores task state information, including a pointer to the task's context
37 * (the task's run time environment, including register values)
38 */
39typedef struct tskTaskControlBlock /* The old naming convention is used to prevent breaking kernel aware debuggers. */
40{
41 volatile StackType_t * pxTopOfStack; /*< Points to the location of the last item placed on the tasks stack. THIS MUST BE THE FIRST MEMBER OF THE TCB STRUCT. */
42
43 #if ( portUSING_MPU_WRAPPERS == 1 )
44 xMPU_SETTINGS xMPUSettings; /*< The MPU settings are defined as part of the port layer. THIS MUST BE THE SECOND MEMBER OF THE TCB STRUCT. */
45 #endif
46
47 ListItem_t xStateListItem; /*< The list that the state list item of a task is reference from denotes the state of that task (Ready, Blocked, Suspended ). */
48 ListItem_t xEventListItem; /*< Used to reference a task from an event list. */
49 UBaseType_t uxPriority; /*< The priority of the task. 0 is the lowest priority. */
50 StackType_t * pxStack; /*< Points to the start of the stack. */
51 char pcTaskName[ configMAX_TASK_NAME_LEN ]; /*< Descriptive name given to the task when created. Facilitates debugging only. */ /*lint !e971 Unqualified char types are allowed for strings and single characters only. */
52
53 #if ( ( portSTACK_GROWTH > 0 ) || ( configRECORD_STACK_HIGH_ADDRESS == 1 ) )
54 StackType_t * pxEndOfStack; /*< Points to the highest valid address for the stack. */
55 #endif
56
57 #if ( portCRITICAL_NESTING_IN_TCB == 1 )
58 UBaseType_t uxCriticalNesting; /*< Holds the critical section nesting depth for ports that do not maintain their own count in the port layer. */
59 #endif
60
61 #if ( configUSE_TRACE_FACILITY == 1 )
62 UBaseType_t uxTCBNumber; /*< Stores a number that increments each time a TCB is created. It allows debuggers to determine when a task has been deleted and then recreated. */
63 UBaseType_t uxTaskNumber; /*< Stores a number specifically for use by third party trace code. */
64 #endif
65
66 #if ( configUSE_MUTEXES == 1 )
67 UBaseType_t uxBasePriority; /*< The priority last assigned to the task - used by the priority inheritance mechanism. */
68 UBaseType_t uxMutexesHeld;
69 #endif
70
71 #if ( configUSE_APPLICATION_TASK_TAG == 1 )
72 TaskHookFunction_t pxTaskTag;
73 #endif
74
75 #if ( configNUM_THREAD_LOCAL_STORAGE_POINTERS > 0 )
76 void * pvThreadLocalStoragePointers[ configNUM_THREAD_LOCAL_STORAGE_POINTERS ];
77 #endif
78
79 #if ( configGENERATE_RUN_TIME_STATS == 1 )
80 uint32_t ulRunTimeCounter; /*< Stores the amount of time the task has spent in the Running state. */
81 #endif
82
83 #if ( configUSE_NEWLIB_REENTRANT == 1 )
84
85 /* Allocate a Newlib reent structure that is specific to this task.
86 * Note Newlib support has been included by popular demand, but is not
87 * used by the FreeRTOS maintainers themselves. FreeRTOS is not
88 * responsible for resulting newlib operation. User must be familiar with
89 * newlib and must provide system-wide implementations of the necessary
90 * stubs. Be warned that (at the time of writing) the current newlib design
91 * implements a system-wide malloc() that must be provided with locks.
92 *
93 * See the third party link http://www.nadler.com/embedded/newlibAndFreeRTOS.html
94 * for additional information. */
95 struct _reent xNewLib_reent;
96 #endif
97
98 #if ( configUSE_TASK_NOTIFICATIONS == 1 )
99 volatile uint32_t ulNotifiedValue[ configTASK_NOTIFICATION_ARRAY_ENTRIES ];
100 volatile uint8_t ucNotifyState[ configTASK_NOTIFICATION_ARRAY_ENTRIES ];
101 #endif
102
103 /* See the comments in FreeRTOS.h with the definition of
104 * tskSTATIC_AND_DYNAMIC_ALLOCATION_POSSIBLE. */
105 #if ( tskSTATIC_AND_DYNAMIC_ALLOCATION_POSSIBLE != 0 ) /*lint !e731 !e9029 Macro has been consolidated for readability reasons. */
106 uint8_t ucStaticallyAllocated; /*< Set to pdTRUE if the task is a statically allocated to ensure no attempt is made to free the memory. */
107 #endif
108
109 #if ( INCLUDE_xTaskAbortDelay == 1 )
110 uint8_t ucDelayAborted;
111 #endif
112
113 #if ( configUSE_POSIX_ERRNO == 1 )
114 int iTaskErrno;
115 #endif
116} tskTCB;
